 vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        
        vector<int> ans;
        vector<int> v;
        int i=0,j=0;
        int n = nums.size();

        while(j<n)
        {
            v.push_back(nums[j]);
            if( j-i +1 < k)
                j++;

            else if((j-i +1)==k)
            {
                ans.push_back( *max_element(v.begin(),v.end()));
                v.erase(v.begin());
                i++;
                j++;
            }    
        }
        return ans;
    }
