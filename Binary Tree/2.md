vector<int> v;
    vector<int> preorderTraversal(TreeNode* root) {

        if(root==NULL)
            return v;

        v.push_back(root->val);
        v=preorderTraversal(root->left);
        v=preorderTraversal(root->right);  
        return v;
    }
