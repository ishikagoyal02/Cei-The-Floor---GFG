int longestOnes(vector<int>& nums, int k) {
        
        int ans=0;
        int i=0,j=0;
        unordered_map<int,int> map;

        while(j< nums.size())
        {
            map[nums[j]]++;

            

            if(map[0]>k)
            {
                while(map[0]> k)
                {
                    map[nums[i]]--;
                    if(map[nums[i]]==0)
                        map.erase(nums[i]);
                    i++;    
                }
            }
            ans= max(ans,j-i+1);
            j++;
        }
        return ans;
    }
