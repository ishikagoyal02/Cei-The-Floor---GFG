vector<int> findAnagrams(string s, string p) {
        
        if(s.size() < p.size()) return {};

        unordered_map<char,int>m;
        unordered_map<char,int> m1;
        vector<int> v;
        int k = p.size();
        int n = s.size();

        cout<<"k : "<<k<<" n: "<<n;
        int i=0,j=0;

        for(int t=0;t<k;t++)
            m[p[t]]++;

        while(j<n)
        {
            m1[s[j]]++;

            if(j-i+1 < k)
                j++;

            else if((j-i+1)==k)
            {
                if(m == m1)
                {
                    v.push_back(i);
                }
                
                    m1[s[i]]--;
                    if(m1[s[i]]==0)
                        m1.erase(s[i]);
            
                i++;
                j++;
            }    
        }
        return v;    
    }
