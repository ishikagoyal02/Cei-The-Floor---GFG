string reverseWords(string S) 
    { 
        
        string  ans="";
        string temp="";
        
        int i=S.size()-1;
        while(i>=0)
        {
            if(S[i]=='.')
            {
                reverse(temp.begin(),temp.end());
                if(ans.empty())
                    ans+=temp;
                else
                {
                    ans=ans+"."+temp;
                }
                temp="";
                i--;
            }
            else
            {
                temp+=S[i];
                i--;
            }
        }
        
        reverse(temp.begin(),temp.end());
                if(ans.empty())
                    ans+=temp;
                else    
                    ans=ans+"."+temp;
        return ans;            
    }
