  void bitManipulation(int num, int i) {
        if(num & (1<<(i-1)))
            cout<<"1 ";
        else
            cout<<"0 ";
            
        
        num=num | (1<<(i-1));           //always 1 at i place
        cout<<num<<" ";
        
        num=num & (~(1<<(i-1)));
        cout<<num;
    }
