int orangesRotting(vector<vector<int>>& grid) {
        
        int n=grid.size();
        int m=grid[0].size();

        int visited[n][m];
        queue<pair<pair<int,int>,int>>q;

        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(grid[i][j]==2)
                {
                    visited[i][j]=2;
                    q.push({{i,j},0});
                }
                else
                {
                    visited[i][j]=0;
                }
            }
        }

        int drow[]={-1,0,1,0};
        int dcol[]={0,1,0,-1};
        int tm=0;

        while(!q.empty())
        {
            int r=q.front().first.first;
            int c=q.front().first.second;

            int t=q.front().second;

            q.pop();

            tm=max(t,tm);

            for(int i=0;i<4;i++)
            {
                int ro=r+drow[i];
                int co=c+dcol[i];

                if(ro>=0 && ro<n && co>=0 && co<m && grid[ro][co]==1 && visited[ro][co]!=2)
                {
                    visited[ro][co]=2;
                    q.push({{ro,co},t+1});
                }
            }
        }

        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
                if(visited[i][j]!=2 && grid[i][j]==1)
                    return -1;
        }

        return tm;
    }
