vector<int> findErrorNums(vector<int>& nums) {
        
        sort(nums.begin(),nums.end());
        int n=nums.size();
        vector<int>ans;
        int t=0,flag=1;
        for(int i=0;i<n-1;i++)
            if(nums[i]==nums[i+1])
                 ans.push_back(nums[i]);
        
        for(int i=1;i<n+1;i++)
            if(find(nums.begin(),nums.end(),i)==nums.end())
                ans.push_back(i);
        
        return ans;
    }
